//mud--a brown variant of the "sludge" that better fits Doom's brown sludgy textures
ACTOR MudChunk
{
	Radius 2
	Height 4
	+NOBLOCKMAP 
	+MISSILE 
	+DROPOFF
	+NOTELEPORT
	+LOWGRAVITY
	+CANNOTPUSH
	+DONTSPLASH
	States
	{
	Spawn:
		MUDS ABCD 8
		Stop
	Death:
		MUDS D 6
		Stop
	}
}

ACTOR MudSplash
{
	+NOBLOCKMAP
	+NOCLIP
	+NOGRAVITY
	+DONTSPLASH
	States
	{
	Spawn:
		MUDS EFGH 6
		Stop
	}
}

actor KXWaterSplash : WaterSplash Replaces WaterSplash
{
    PROJECTILE
    -NOGRAVITY
    +NOBLOCKMAP
    +CLIENTSIDEONLY
	+ALLOWPARTICLES
	+THRUACTORS
	+CANNOTPUSH
    +DONTSPLASH
    +DONTBLAST
	Gravity 1.0
    mass 5
	Health 1
    States
    {
    Spawn:
        SPSH A 2 
        loop
	Death:
        SPSH ABC 4
        SPSH D 8
        Stop
    }
}

  +NOBLOCKMAP
